class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int sum = 0;
        int count = 0;
        unordered_map<int, int> mappy;  //sum of subarray --> count of subarrays
        mappy[0] = 1;

        for(int i = 0; i < nums.size(); i++)
        {
            sum += nums[i];
            count += mappy[sum - k];
            mappy[sum]++;
        }
        return count;
    }
};
